folium_map = folium.Map(
    location=(city_df.lat.median(), city_df.lng.median()),
    zoom_start=2)

popup_text = ("{}<br>"
              "Count: {}<br>")

for idx, row in city_df.iterrows():
    folium.CircleMarker(
        location=(row.lat, row.lng),
        radius=row.cnt,
        color="green",
        fill=True,
        popup=folium.Popup(
            html=popup_text.format(row.city, row.cnt),
            max_width=1000) 
    ).add_to(folium_map)
