folium_map = folium.Map(tiles='stamenwatercolor')

popup_text = ("{}<br>"
              "Count: {}<br>"
              "Songs: {}<br>")

for idx, row in city_df.iterrows():
    folium.CircleMarker(
        location=(row.lat, row.lng),
        radius=row.cnt*1.3,
        color="purple",
        fill=True,
        popup=popup_text.format(
            row.city, row.cnt, 'songs..')
    ).add_to(folium_map)
